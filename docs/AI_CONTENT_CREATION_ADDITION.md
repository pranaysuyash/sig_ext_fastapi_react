### Batch Generation Script

**Python script using OpenAI API to generate 50 descriptions:**

```python
import openai
import csv
import time
import os

# Configuration
openai.api_key = os.getenv("OPENAI_API_KEY")

# Define all 50 directories with context
directories = [
    {"name": "Product Hunt", "focus": "tech startups", "audience": "developers, founders", "max_words": 200},
    {"name": "Capterra", "focus": "business software", "audience": "business buyers", "max_words": 250},
    # ... add all 50
]

master_description = """
SignKit: Privacy-first desktop signature extraction and PDF signing.
- Sub-1-second extraction using computer vision
- 100% local processing, zero cloud
- One-time purchase ($29-299)
- Targets: freelancers, healthcare, legal
- Built by ex-EY consultant
"""

results = []

for directory in directories:
    prompt = f"""
Adapt this for {directory['name']} (focuses on {directory['focus']}, targets {directory['audience']}).

Master: {master_description}

Max {directory['max_words']} words. Write unique, compelling description.
"""
    
    response = openai.ChatCompletion.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": prompt}],
        max_tokens=500
    )
    
    results.append({
        "Directory": directory['name'],
        "Description": response.choices[0].message.content
    })
    
    time.sleep(0.5)

# Export to CSV
with open('signkit_descriptions.csv', 'w', newline='') as f:
    writer = csv.DictWriter(f, fieldnames=['Directory', 'Description'])
    writer.writeheader()
    writer.writerows(results)

print(f"âœ“ Generated {len(results)} descriptions!")
print(f"Total cost: ~${len(directories) * 0.001:.2f}")
print(f"vs Manual: ~25 hours")
```

**Output:** 50 unique descriptions in 25 minutes for ~$0.05
